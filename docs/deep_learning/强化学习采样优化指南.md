## 基础概念

### 什么是采样优化

采样优化是指在有限资源（预算、时间等）约束下，智能选择采样点位置，以最大化获取的信息价值或最小化不确定性。

**核心要素**：
- **不确定性地图**：表示空间各点的不确定性分布
- **采样预算**：允许的最大采样点数量
- **采样策略**：决定采样点选择规则的函数
- **评估指标**：衡量采样效果的指标（不确定性降低、精度提升等）

**传统方法 vs 强化学习**：

| 方法 | 优势 | 劣势 |
|------|------|------|
| 随机采样 | 简单，无偏 | 效率低，不考虑空间相关性 |
| 网格采样 | 覆盖均匀 | 不考虑不确定性分布 |
| 贪心采样 | 直观，效率较高 | 容易陷入局部最优 |
| 克里金采样 | 考虑空间相关性 | 计算复杂，假设性强 |
| **强化学习** | **自适应，考虑全局** | **需要训练，计算成本高** |

### 为什么使用强化学习

强化学习在采样优化中具有独特优势：

**1. 自适应决策**
- 根据当前不确定性分布动态调整策略
- 考虑长期累积奖励，而非局部最优

**2. 多目标平衡**
- 同时优化不确定性降低、精度提升、空间覆盖等多个目标
- 自动权衡不同目标的优先级

**3. 约束处理**
- 自然融入预算约束、空间约束等
- 在约束内寻找最优解

**4. 持续学习**
- 支持在线学习，适应环境变化
- 可以利用历史经验持续改进

**1. 环境监测**

**2. 地质勘探**

#### 2. 自定义状态设计

```python
class CustomSamplingEnv(SamplingEnv):
    """自定义采样环境"""

    def _observation(self) -> dict[str, np.ndarray]:
        # 获取基础观测
        obs = super()._observation()

        # 添加自定义特征
        obs["gradient_map"] = self._compute_gradient()
        obs["local_variance"] = self._compute_local_variance()
        obs["nearest_distance"] = self._compute_nearest_distance()

        return obs

    def _compute_gradient(self) -> np.ndarray:
        """计算不确定性梯度"""
        grad_y, grad_x = np.gradient(self.current_uncertainty)
        return np.sqrt(grad_x**2 + grad_y**2)

    def _compute_local_variance(self) -> np.ndarray:
        """计算局部方差"""
        from scipy.ndimage import uniform_filter
        local_mean = uniform_filter(self.current_uncertainty, size=5)
        local_sq_mean = uniform_filter(self.current_uncertainty**2, size=5)
        return local_sq_mean - local_mean**2

    def _compute_nearest_distance(self) -> np.ndarray:
        """计算到最近采样点的距离"""
        if not self.sampled_points:
            return np.ones_like(self.current_uncertainty)

        # 计算距离场
        distance_map = np.full_like(self.current_uncertainty, np.inf)
        for i, j in self.sampled_indices:
            rr, cc = np.indices(distance_map.shape)
            dist = np.sqrt((rr - i)**2 + (cc - j)**2)
            distance_map = np.minimum(distance_map, dist)

        return distance_map
```

```python
class CustomSamplingEnv(SamplingEnv):
    """自定义动作解码"""

    def _decode_action(self, action: Any) -> tuple[list[tuple[int, int]], float, float]:
        """自定义动作解码逻辑"""

        if self.action_mode == "region_based":
            # 区域采样：指定矩形区域
            x, y, width, height = action
            points = self._sample_in_region(x, y, width, height)

        elif self.action_mode == "adaptive":
            # 自适应采样：基于不确定性选择多个点
            n_points = action
            points = self._sample_adaptive(n_points)

        else:
            # 使用默认解码
            points, boundary_pen, distance_pen = super()._decode_action(action)
            return points, boundary_pen, distance_pen

        # 计算约束惩罚
        boundary_pen = self._compute_boundary_penalty(points)
        distance_pen = self._compute_distance_penalty(points)

        return points, boundary_pen, distance_pen

    def _sample_in_region(self, x, y, width, height):
        """在区域内采样"""
        min_x, max_x, min_y, max_y = self.boundary
        col_start = int((x - width/2) / max(max_x - min_x, 1e-8) * self.w)
        col_end = int((x + width/2) / max(max_x - min_x, 1e-8) * self.w)
        row_start = int((y - height/2) / max(max_y - min_y, 1e-8) * self.h)
        row_end = int((y + height/2) / max(max_y - min_y, 1e-8) * self.h)

        points = []
        for row in range(max(0, row_start), min(self.h, row_end)):
            for col in range(max(0, col_start), min(self.w, col_end)):
                points.append((row, col))

        return points

    def _sample_adaptive(self, n_points):
        """基于不确定性自适应采样"""
        flat_uncertainty = self.current_uncertainty.flatten()
        top_indices = np.argpartition(flat_uncertainty, -n_points)[-n_points:]
        points = [divmod(idx, self.w) for idx in top_indices]
        return points
```

#### 3. 自定义奖励函数

```python
class CustomRewardComposer:
    """自定义奖励组合器"""

    def __init__(self, weights):
        self.weights = weights

    def compose(self, uncertainty_reduction, accuracy_improvement,
                spatial_coverage, sampling_cost_penalty,
                boundary_penalty, distance_penalty, **kwargs):

        # 基础奖励
        base_reward = (
            self.weights.uncertainty_reduction * uncertainty_reduction +
            self.weights.accuracy_improvement * accuracy_improvement +
            self.weights.spatial_coverage * spatial_coverage
        )

        # 惩罚项
        penalty = (
            self.weights.sampling_cost * sampling_cost_penalty +
            self.weights.boundary_constraint * boundary_penalty +
            self.weights.distance_constraint * distance_penalty
        )

        # 奖励塑形：鼓励探索新区域
        exploration_bonus = kwargs.get("exploration_bonus", 0.0)

        # 奖励塑形：惩罚聚集
        clustering_penalty = kwargs.get("clustering_penalty", 0.0)

        # 总奖励
        total_reward = base_reward - penalty + exploration_bonus - clustering_penalty

        return {
            "base_reward": base_reward,
            "penalty": penalty,
            "exploration_bonus": exploration_bonus,
            "clustering_penalty": clustering_penalty,
            "total_reward": total_reward
        }
```

#### 4. 奖励塑形技巧

```python
class PotentialBasedRewardShaper:
    """基于势能的奖励塑形"""

    def __init__(self, potential_func, gamma=0.99):
        self.potential_func = potential_func
        self.gamma = gamma

    def shape(self, reward, state, next_state):
        """应用奖励塑形"""
        phi_t = self.potential_func(state)
        phi_t1 = self.potential_func(next_state)

        shaped_reward = reward + self.gamma * phi_t1 - phi_t
        return shaped_reward

# 示例势能函数
def uncertainty_potential(observation):
    """基于不确定性的势能"""
    return -np.mean(observation["uncertainty_map"])

def coverage_potential(observation):
    """基于覆盖率的势能"""
    coverage = np.mean(observation["sampling_distribution"])
    return coverage

def diversity_potential(observation):
    """基于多样性的势能"""
    sampled_values = observation["sampled_values"]
    if np.sum(sampled_values) == 0:
        return 0.0
    return np.std(sampled_values[sampled_values > 0])
```

```python
def generate_synthetic_data(h=50, w=50, n_peaks=5, seed=42):
    """生成合成不确定性地图"""
    rng = np.random.default_rng(seed)

    # 基础不确定性
    uncertainty_map = rng.random((h, w))

    # 添加高斯峰
    for _ in range(n_peaks):
        peak_x = rng.integers(0, w)
        peak_y = rng.integers(0, h)
        sigma = rng.uniform(5, 15)
        amplitude = rng.uniform(0.3, 0.7)

        xx, yy = np.meshgrid(np.arange(w), np.arange(h))
        peak = amplitude * np.exp(-((xx - peak_x)**2 + (yy - peak_y)**2) / (2 * sigma**2))
        uncertainty_map += peak

    # 归一化
    uncertainty_map = (uncertainty_map - uncertainty_map.min()) / (uncertainty_map.max() - uncertainty_map.min())

    return uncertainty_map
```

```python
def single_point_inference(agent, uncertainty_map, position):
    """单点推理：评估指定位置的采样价值"""

    # 准备观测
    h, w = uncertainty_map.shape
    env = SamplingEnv(
        uncertainty_map=uncertainty_map,
        action_mode="discrete",
        budget=20
    )

    obs = env.reset()

    # 计算动作索引
    x, y = position
    col = int(np.clip(x / (env.boundary[1] - env.boundary[0]) * (w - 1), 0, w - 1))
    row = int(np.clip(y / (env.boundary[3] - env.boundary[2]) * (h - 1), 0, h - 1))
    action_idx = row * w + col

    # 获取预测
    if isinstance(agent, PPOAgent):
        pred = agent.predict_ppo([obs], deterministic=True)
        action_prob = pred["action_probabilities"][0][action_idx]
        state_value = pred["state_values"][0]
    elif isinstance(agent, DQNAgent):
        state = flatten_observation(obs)
        q_values = agent._q_values(state)
        action_prob = _softmax(q_values / agent.epsilon)[action_idx]
        state_value = float(np.max(q_values))

    return {
        "position": position,
        "action_index": action_idx,
        "action_probability": action_prob,
        "state_value": state_value,
        "recommendation": "high" if action_prob > 0.8 else "medium" if action_prob > 0.5 else "low"
    }
```

### 批量推理

```python
def batch_inference(agent, uncertainty_map, candidate_positions):
    """批量推理：评估多个候选位置"""

    h, w = uncertainty_map.shape
    env = SamplingEnv(
        uncertainty_map=uncertainty_map,
        action_mode="discrete",
        budget=20
    )

    obs = env.reset()

    # 计算所有候选点的动作索引
    action_indices = []
    for x, y in candidate_positions:
        col = int(np.clip(x / (env.boundary[1] - env.boundary[0]) * (w - 1), 0, w - 1))
        row = int(np.clip(y / (env.boundary[3] - env.boundary[2]) * (h - 1), 0, h - 1))
        action_indices.append(row * w + col)

    # 批量预测
    if isinstance(agent, PPOAgent):
        pred = agent.predict_ppo([obs] * len(candidate_positions), deterministic=True)

        results = []
        for i, action_idx in enumerate(action_indices):
            results.append({
                "position": candidate_positions[i],
                "action_index": action_idx,
                "action_probability": pred["action_probabilities"][i][action_idx],
                "state_value": pred["state_values"][i],
                "entropy": pred["policy_entropy"][i]
            })

    elif isinstance(agent, DQNAgent):
        state = flatten_observation(obs)
        q_values = agent._q_values(state)
        probs = _softmax(q_values / agent.epsilon)

        results = []
        for i, action_idx in enumerate(action_indices):
            results.append({
                "position": candidate_positions[i],
                "action_index": action_idx,
                "action_probability": probs[action_idx],
                "state_value": float(q_values[action_idx]),
                "entropy": -np.sum(probs * np.log(probs + 1e-12))
            })

    # 按概率排序
    results.sort(key=lambda x: x["action_probability"], reverse=True)

    return results
```

```python
class OnlineSamplingOptimizer:
    """在线采样优化器"""

    def __init__(self, agent, env, update_interval=5):
        self.agent = agent
        self.env = env
        self.update_interval = update_interval
        self.step_count = 0
        self.obs_history = []
        self.action_history = []
        self.reward_history = []

    def recommend_sample(self, current_observation):
        """推荐下一个采样点"""
        # 使用当前策略推荐
        if isinstance(self.agent, PPOAgent):
            action_idx, log_prob, value = self.agent.select_action(
                current_observation,
                deterministic=True
            )
        elif isinstance(self.agent, DQNAgent):
            action_idx = self.agent.select_action(
                current_observation,
                deterministic=True
            )

        return _action_from_discrete(action_idx, self.env)

    def update(self, observation, action, reward, next_observation, done):
        """在线更新"""
        self.step_count += 1

        # 存储经验
        if isinstance(self.agent, PPOAgent):
            # 需要重新获取log_prob和value
            action_idx, log_prob, value = self.agent.select_action(
                observation,
                deterministic=False
            )
            self.agent.store_transition(
                observation, action_idx, reward, done, log_prob, value
            )
        elif isinstance(self.agent, DQNAgent):
            action_idx = _action_to_index(action, self.env)
            self.agent.store_transition(
                observation, action_idx, reward, next_observation, done
            )

        # 定期更新
        if self.step_count % self.update_interval == 0:
            if isinstance(self.agent, PPOAgent):
                self.agent.update(last_value=self.agent.value(next_observation))
            elif isinstance(self.agent, DQNAgent):
                self.agent.train_step()

        # 记录历史
        self.obs_history.append(observation)
        self.action_history.append(action)
        self.reward_history.append(reward)

        # 限制历史长度
        if len(self.obs_history) > 1000:
            self.obs_history.pop(0)
            self.action_history.pop(0)
            self.reward_history.pop(0)

    def get_performance_stats(self):
        """获取性能统计"""
        return {
            "total_steps": self.step_count,
            "mean_reward": np.mean(self.reward_history) if self.reward_history else 0.0,
            "recent_rewards": self.reward_history[-10:] if len(self.reward_history) >= 10 else self.reward_history
        }
```

```python
    """增强的特征工程器"""

    def __init__(self):
        self.base_engineer = SamplingFeatureEngineer()

    def extract_features(self, observation, action, action_space):
        """提取增强特征"""

        # 基础特征
        base_features = self.base_engineer.extract_state_action_features(
            observation, action, action_space
        )

        # 添加自定义特征
        custom_features = self._extract_custom_features(observation, action)

        # 合并特征
        combined = np.concatenate([
            base_features.feature_vector,
            custom_features
        ])

        return combined

    def _extract_custom_features(self, observation, action):
        """提取自定义特征"""
        uncertainty_map = observation["uncertainty_map"]
        sampled_mask = observation["sampling_distribution"]

        # 局部不确定性特征
        local_mean = np.mean(uncertainty_map)
        local_std = np.std(uncertainty_map)
        local_max = np.max(uncertainty_map)

        # 采样密度特征
        sampled_count = np.sum(sampled_mask > 0)
        total_count = sampled_mask.size
        sampling_density = sampled_count / total_count if total_count > 0 else 0.0

        # 空间分布特征
        if sampled_count > 0:
            sampled_coords = np.where(sampled_mask > 0)
            center_of_mass = np.array([
                np.mean(sampled_coords[1]),
                np.mean(sampled_coords[0])
            ])
        else:
            center_of_mass = np.array([uncertainty_map.shape[1] / 2, uncertainty_map.shape[0] / 2])

        return np.array([
            local_mean,
            local_std,
            local_max,
            sampling_density,
            center_of_mass[0],
            center_of_mass[1]
        ])
```

```python
class PrioritizedReplayBuffer:
    """优先经验回放缓冲区"""

    def __init__(self, capacity=10000, alpha=0.6, beta=0.4):
        self.capacity = capacity
        self.alpha = alpha  # 优先级指数
        self.beta = beta    # 重要性采样指数
        self.buffer = []
        self.priorities = []
        self.pos = 0

    def add(self, transition, priority=None):
        """添加经验"""
        if priority is None:
            priority = 1.0

        if len(self.buffer) < self.capacity:
            self.buffer.append(transition)
            self.priorities.append(priority)
        else:
            self.buffer[self.pos] = transition
            self.priorities[self.pos] = priority

        self.pos = (self.pos + 1) % self.capacity

    def sample(self, batch_size):
        """采样经验"""
        if len(self.buffer) == 0:
            return [], [], []

        # 计算采样概率
        priorities = np.array(self.priorities)
        probs = priorities ** self.alpha
        probs /= probs.sum()

        # 采样索引
        indices = np.random.choice(
            len(self.buffer),
            size=min(batch_size, len(self.buffer)),
            replace=False,
            p=probs
        )

        # 计算重要性权重
        weights = (len(self.buffer) * probs[indices]) ** (-self.beta)
        weights /= weights.max()

        # 获取样本
        samples = [self.buffer[i] for i in indices]

        return samples, indices, weights

    def update_priorities(self, indices, priorities):
        """更新优先级"""
        for idx, priority in zip(indices, priorities):
            self.priorities[idx] = priority
```

```python
    """并行训练"""
    with ProcessPoolExecutor(max_workers=n_workers) as executor:
        futures = [
            executor.submit(
                train_worker,
                env_config,
                agent_config,
                seed=i
            )
            for i in range(n_workers)

    return {
        "mean_reward": np.mean(rewards),
        "std_reward": np.std(rewards),
        "best_reward": np.max(rewards),

    }
```

```python
class MultiObjectiveSamplingOptimizer:
    """多目标采样优化器"""

    def __init__(self, objectives):
        self.objectives = objectives
        self.pareto_front = []

    def evaluate(self, sampling_strategy):
        """评估采样策略"""
        results = {}
        for name, objective in self.objectives.items():
            results[name] = objective.evaluate(sampling_strategy)
        return results

    def is_dominated(self, solution1, solution2):
        """判断solution1是否被solution2支配"""
        better_in_any = False
        for obj_name in self.objectives:
            if solution1[obj_name] > solution2[obj_name]:
                better_in_any = True
            elif solution1[obj_name] < solution2[obj_name]:
                return False
        return better_in_any

    def update_pareto_front(self, new_solution):
        """更新帕累托前沿"""
        # 检查是否被现有解支配
        dominated = False
        for existing in self.pareto_front:
            if self.is_dominated(new_solution, existing):
                dominated = True
                break

        if not dominated:
            # 移除被新解支配的解
            self.pareto_front = [
                sol for sol in self.pareto_front
                if not self.is_dominated(sol, new_solution)
            ]
            self.pareto_front.append(new_solution)

    def select_solution(self, weights):
        """根据权重选择解"""
        if not self.pareto_front:
            return None

        # 加权评分
        best_solution = None
        best_score = float('-inf')

        for solution in self.pareto_front:
            score = sum(
                weights.get(obj, 1.0) * solution[obj]
                for obj in self.objectives
            )
            if score > best_score:
                best_score = score
                best_solution = solution

        return best_solution
```

```python
class ConstrainedSamplingOptimizer:
    """约束采样优化器"""

    def __init__(self, env, agent, constraints):
        self.env = env
        self.agent = agent
        self.constraints = constraints

    def check_constraints(self, action):
        """检查约束"""
        for constraint in self.constraints:
            if not constraint.check(action):
                return False
        return True

    def select_constrained_action(self, observation):
        """选择满足约束的动作"""
        # 获取候选动作
        if isinstance(self.agent, PPOAgent):
            pred = self.agent.predict_ppo([observation], deterministic=False)
            candidates = list(enumerate(pred["action_probabilities"][0]))
        elif isinstance(self.agent, DQNAgent):
            state = flatten_observation(observation)
            q_values = self.agent._q_values(state)
            candidates = list(enumerate(q_values))

        # 按价值/概率排序
        candidates.sort(key=lambda x: x[1], reverse=True)

        # 找到第一个满足约束的动作
        for action_idx, value in candidates:
            action = _action_from_discrete(action_idx, self.env)
            if self.check_constraints(action):
                return action

        # 如果没有满足约束的动作，返回随机动作
        return self.env.sample_random_action()

# 示例约束
class BudgetConstraint:
    def __init__(self, budget):
        self.budget = budget
        self.used = 0

    def check(self, action):
        """检查预算约束"""
        # 简单实现：假设每个动作消耗1个预算
        return self.used < self.budget

    def update(self, action):
        """更新已用预算"""
        self.used += 1

class DistanceConstraint:
    def __init__(self, min_distance, sampled_points):
        self.min_distance = min_distance
        self.sampled_points = sampled_points

    def check(self, action):
        """检查距离约束"""
        if not self.sampled_points:
            return True

        # 计算到已有采样点的最小距离
        new_x, new_y = action
        min_dist = min(
            np.hypot(new_x - x, new_y - y)
            for x, y in self.sampled_points
        )

        return min_dist >= self.min_distance
```

```python
class AdaptiveSamplingOptimizer:
    """自适应采样优化器"""

    def __init__(self, agent, env, adaptation_threshold=0.1):
        self.agent = agent
        self.env = env
        self.adaptation_threshold = adaptation_threshold
        self.performance_history = []
        self.last_performance = None

    def detect_drift(self):
        """检测性能漂移"""
        if len(self.performance_history) < 10:
            return False

        recent_performance = np.mean(self.performance_history[-10:])
        if self.last_performance is None:
            self.last_performance = recent_performance
            return False

        drift = abs(recent_performance - self.last_performance) / (self.last_performance + 1e-8)

        return drift > self.adaptation_threshold

    def adapt(self):
        """适应环境变化"""
        print("Detected performance drift, adapting...")

        # 重置环境
        obs = self.env.reset()

        # 快速重新训练
        for episode in range(10):
            done = False
            while not done:
                if isinstance(self.agent, PPOAgent):
                    action_idx, log_prob, value = self.agent.select_action(obs)
                    next_obs, reward, done, _ = self.env.step(
                        _action_from_discrete(action_idx, self.env)
                    )
                    self.agent.store_transition(obs, action_idx, reward, done, log_prob, value)
                obs = next_obs

            self.agent.update()

        # 更新性能基线
        self.last_performance = np.mean(self.performance_history[-10:])

    def recommend_with_adaptation(self, observation):
        """带自适应的推荐"""
        # 检查是否需要适应
        if self.detect_drift():
            self.adapt()

        # 正常推荐
        action_idx, _, _ = self.agent.select_action(observation, deterministic=True)
        return _action_from_discrete(action_idx, self.env)
```
