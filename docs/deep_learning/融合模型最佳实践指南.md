# 融合模型最佳实践指南

## 概述

本指南提供了使用 UDAKE 融合模型系统的最佳实践，涵盖了从设计、实施到优化的完整流程。遵循这些实践可以帮助您充分发挥融合模型的潜力，提高预测性能和系统稳定性。

## 设计原则

### 1. 模型多样性原则

**原则**：参与融合的子模型应具有足够的多样性，以捕捉数据的不同特征和模式。

**实践**：
- 使用不同架构的模型（如CNN、RNN、Transformer）
- 使用不同训练策略（如不同损失函数、优化器）
- 使用不同特征表示（如原始特征、嵌入特征）
- 使用不同超参数配置

**示例**：
```python
from deep_learning.fusion import ModelPrediction

# 不同架构的模型
models = [
    ModelPrediction(
        model_id="cnn_model",
        predictions=cnn_predictions,
        variances=cnn_variances
    ),
    ModelPrediction(
        model_id="rnn_model",
        predictions=rnn_predictions,
        variances=rnn_variances
    ),
    ModelPrediction(
        model_id="transformer_model",
        predictions=transformer_predictions,
        variances=transformer_variances
    )
]
```

**验证**：
```python
import numpy as np

# 检查模型相关性
corr_matrix = np.corrcoef([
    model.predictions for model in models
])
print("模型相关性矩阵:")
print(corr_matrix)

# 理想情况：相关性适中（0.5-0.8）
# 避免过高相关性（>0.9）：模型冗余
# 避免过低相关性（<0.3）：模型可能无效
```

### 2. 性能互补原则

**原则**：子模型应在不同场景下表现良好，形成互补优势。

**实践**：
- 分析每个模型的强项和弱项
- 识别不同数据分布下的最优模型
- 结合局部和全局模型
- 结合快慢模型

**示例**：
```python
# 分析模型在不同区域的性能
def analyze_regional_performance(models, true_values, regions):
    regional_performance = {}
    for region in regions:
        region_mask = regions == region
        for model in models:
            pred = np.array(model.predictions)[region_mask]
            true = np.array(true_values)[region_mask]
            rmse = np.sqrt(np.mean((pred - true) ** 2))
            regional_performance.setdefault(region, {})[model.model_id] = rmse
    return regional_performance

performance = analyze_regional_performance(models, true_values, regions)
for region, perf in performance.items():
    best_model = min(perf, key=perf.get)
    print(f"区域 {region} 最优模型: {best_model}")
```

### 3. 渐进式集成原则

**原则**：从简单策略开始，逐步引入复杂策略。

**实践步骤**：
1. **第一阶段**：使用简单平均，建立基线
2. **第二阶段**：引入加权平均，评估性能提升
3. **第三阶段**：尝试中位数、最大最小等策略
4. **第四阶段**：考虑堆叠融合、动态融合等高级策略

**示例**：
```python
from deep_learning.fusion import FusionConfig, FusionStrategy

# 渐进式测试
strategies = [
    FusionStrategy.SIMPLE_AVERAGE,
    FusionStrategy.WEIGHTED_AVERAGE,

    FusionStrategy.STACKING,

baseline_rmse = float('inf')

    config = FusionConfig(strategy=strategy)

    print(f"{strategy.value}: RMSE={rmse:.4f}, 改进={improvement:.2f}%")
    baseline_rmse = rmse
```

import numpy as np

def check_data_quality(models, true_values):
    """检查数据质量"""
    issues = []

    # 检查预测长度一致性
    lengths = [len(model.predictions) for model in models]
    if len(set(lengths)) > 1:
        issues.append(f"预测长度不一致: {lengths}")

    # 检查缺失值
    for model in models:
        if np.isnan(model.predictions).any():
            issues.append(f"模型 {model.model_id} 包含缺失值")

    # 检查异常值
    for model in models:
        pred = np.array(model.predictions)
        z_scores = np.abs((pred - pred.mean()) / pred.std())
        if (z_scores > 5).any():
            issues.append(f"模型 {model.model_id} 包含极端异常值")

    # 检查真值质量
    if true_values:
        if np.isnan(true_values).any():
            issues.append("真值包含缺失值")

    return issues

issues = check_data_quality(models, true_values)
if issues:
    print("数据质量问题:")
    for issue in issues:
        print(f"  - {issue}")
else:
    print("数据质量检查通过")
```

def standardize_predictions(models, true_values=None):
    """标准化预测结果"""
    all_preds = np.concatenate([model.predictions for model in models])
    mean = all_preds.mean()
    std = all_preds.std()

    standardized_models = []
    for model in models:
        pred = (np.array(model.predictions) - mean) / std
        standardized_models.append(
            ModelPrediction(
                model_id=model.model_id,
                predictions=pred.tolist(),
                variances=model.variances
            )
        )

    standardized_true = None
    if true_values:
        standardized_true = ((np.array(true_values) - mean) / std).tolist()

    return standardized_models, standardized_true, mean, std

def evaluate_single_models(models, true_values):
    """评估单个模型性能"""
    evaluations = []

    for model in models:
        pred = np.array(model.predictions)
        true = np.array(true_values)

        rmse = np.sqrt(np.mean((pred - true) ** 2))
        mae = np.mean(np.abs(pred - true))
        r2 = 1 - np.sum((true - pred) ** 2) / np.sum((true - true.mean()) ** 2)

        evaluations.append({
            'model_id': model.model_id,
            'rmse': rmse,
            'mae': mae,
            'r2': r2
        })

    # 排序
    evaluations.sort(key=lambda x: x['rmse'])

    print("单模型性能评估:")
    for eval in evaluations:
        print(f"  {eval['model_id']}:")
        print(f"    RMSE: {eval['rmse']:.4f}")
        print(f"    MAE: {eval['mae']:.4f}")
        print(f"    R²: {eval['r2']:.4f}")

    return evaluations

evaluations = evaluate_single_models(models, true_values)

    print("模型相关性矩阵:")
    for i, model_i in enumerate(models):

            if i < j:
                corr = corr_matrix[i, j]
                print(f"  {model_i.model_id} vs {model_j.model_id}: {corr:.4f}")

    # 检查高相关性
    high_corr_pairs = []
    for i, model_i in enumerate(models):

        print("\n高相关性模型对（可能冗余）:")
        for pair in high_corr_pairs:

from itertools import product

def grid_search(models, true_values, param_grid):
    """网格搜索最优参数"""
    best_score = float('inf')
    best_params = None
    best_result = None

    # 生成参数组合
    keys = param_grid.keys()
    values = param_grid.values()
    combinations = product(*values)

    total = len(list(product(*values)))
    print(f"开始网格搜索，共 {total} 种组合")

    for i, combination in enumerate(combinations, 1):
        params = dict(zip(keys, combination))

        try:
            config = FusionConfig(**params)
            result = engine.fuse(models, config=config, true_values=true_values)
            score = result.metrics['rmse']

            if score < best_score:
                best_score = score
                best_params = params
                best_result = result

            if i % 10 == 0:
                print(f"进度: {i}/{total}, 当前最优: RMSE={best_score:.4f}")

        except Exception as e:
            print(f"参数组合 {params} 失败: {e}")
            continue

    print(f"\n最优参数:")
    for key, value in best_params.items():
        print(f"  {key}: {value}")
    print(f"最优RMSE: {best_score:.4f}")

    return best_params, best_result

# 定义参数网格
param_grid = {

import random

def bayesian_optimization(models, true_values, n_iterations=50):
    """简化的贝叶斯优化"""
    best_score = float('inf')
    best_params = None

    # 初始随机采样
    for _ in range(10):
        params = {
            'strategy': random.choice([FusionStrategy.WEIGHTED_AVERAGE, FusionStrategy.DYNAMIC]),
            'weight_method': random.choice([WeightMethod.RMSE_BASED, WeightMethod.MAE_BASED]),
            'smoothing': random.choice([True, False]),
            'smoothing_factor': random.uniform(0.05, 0.2),
            'min_weight': random.uniform(0.0, 0.1),
            'max_weight': random.uniform(0.5, 0.8)
        }

        try:
            config = FusionConfig(**params)
            result = engine.fuse(models, config=config, true_values=true_values)
            score = result.metrics['rmse']

            if score < best_score:
                best_score = score
                best_params = params

        except Exception as e:
            continue

    # 基于最佳参数的局部搜索
    for _ in range(n_iterations - 10):
        params = best_params.copy()
        params['smoothing_factor'] += random.uniform(-0.05, 0.05)
        params['smoothing_factor'] = max(0.05, min(0.2, params['smoothing_factor']))

            result = engine.fuse(models, config=config, true_values=true_values)
            score = result.metrics['rmse']

            if score < best_score:
                best_score = score
                best_params = params

        except Exception as e:
            continue

    print(f"最优参数: {best_params}")

class FusionMonitor:
    """融合性能监控器"""

    def __init__(self):
        self.history = []

    def record(self, result, true_values=None):
        """记录融合结果"""
        metrics = result.metrics.copy()
        if true_values:
            pred = np.array(result.fused_predictions)
            true = np.array(true_values)
            metrics['actual_rmse'] = float(np.sqrt(np.mean((pred - true) ** 2)))

        self.history.append({
            'timestamp': time.time(),
            'strategy': result.strategy,
            'weight_method': result.weight_method,
            'metrics': metrics,
            'weights': result.weights,
            'diagnostics': result.diagnostics
        })

    def analyze_trends(self):
        """分析性能趋势"""
        if len(self.history) < 2:
            print("数据不足，无法分析趋势")
            return

        rmse_values = [h['metrics'].get('rmse', h['metrics'].get('actual_rmse', float('inf')))
                      for h in self.history]

        # 计算趋势
        recent_rmse = np.mean(rmse_values[-5:])
        early_rmse = np.mean(rmse_values[:5])
        trend = (recent_rmse - early_rmse) / early_rmse * 100

        print(f"性能趋势分析:")
        print(f"  早期平均RMSE: {early_rmse:.4f}")
        print(f"  近期平均RMSE: {recent_rmse:.4f}")
        print(f"  变化率: {trend:+.2f}%")

        if trend > 5:
            print("  ⚠️  警告: 性能明显下降")
        elif trend < -5:
            print("  ✓  性能明显提升")
        else:
            print("  →  性能稳定")

monitor = FusionMonitor()
monitor.record(result, true_values)
monitor.analyze_trends()
```

def detect_anomalies(result, baseline_metrics, threshold=2.0):
    """检测融合结果异常"""
    anomalies = []

    current_metrics = result.metrics

    for metric, baseline_value in baseline_metrics.items():
        if metric in current_metrics:
            current_value = current_metrics[metric]
            if current_value > baseline_value * threshold:
                anomalies.append({
                    'metric': metric,
                    'current': current_value,
                    'baseline': baseline_value,
                    'ratio': current_value / baseline_value
                })

    if anomalies:
        print("检测到异常:")
        for anomaly in anomalies:
            print(f"  {anomaly['metric']}:")
            print(f"    当前值: {anomaly['current']:.4f}")
            print(f"    基准值: {anomaly['baseline']:.4f}")
            print(f"    比率: {anomaly['ratio']:.2f}x")
    else:
        print("未检测到异常")

    return anomalies

from concurrent.futures import ThreadPoolExecutor

def parallel_inference(engine, models_list, config_list, max_workers=4):
    """并行推理"""
    results = [None] * len(models_list)

    def process_single(idx, models, config):
        results[idx] = engine.fuse(models, config=config)

    with ThreadPoolExecutor(max_workers=max_workers) as executor:
        futures = []
        for idx, (models, config) in enumerate(zip(models_list, config_list)):
            future = executor.submit(process_single, idx, models, config)
            futures.append(future)

        for future in futures:
            future.result()

    return results

# 使用示例

        print(f"权重分布: {result.weights}")
        return True
    except Exception as e:
        print(f"权重计算失败: {e}")
        return False

#### 1.2 融合结果异常
**问题**：融合结果超出预期范围或明显错误

**原因**：
- 模型预测异常
- 权重配置错误
- 融合策略不适用

**解决方案**：
```python
def validate_fusion_result(result, expected_range=None):
    """验证融合结果"""
    print("验证融合结果...")

    fused = np.array(result.fused_predictions)

    # 1. 检查NaN
    if np.isnan(fused).any():
        print("错误: 融合结果包含NaN")
        return False

    # 2. 检查范围
    if expected_range:
        min_val, max_val = expected_range
        out_of_range = (fused < min_val) | (fused > max_val)
        if out_of_range.any():
            print(f"警告: {out_of_range.sum()} 个预测超出预期范围 [{min_val}, {max_val}]")

    # 3. 检查统计特性
    print(f"融合结果统计:")
    print(f"  均值: {fused.mean():.4f}")
    print(f"  标准差: {fused.std():.4f}")
    print(f"  范围: [{fused.min():.4f}, {fused.max():.4f}]")

    return True

# 使用示例
validate_fusion_result(result, expected_range=(0, 1))
```

#### 1.3 性能退化
**问题**：融合性能随时间下降

**原因**：
- 数据分布变化
- 模型性能退化
- 权重配置过时

**解决方案**：
```python
def detect_performance_degradation(monitor, window_size=10, threshold=0.1):
    """检测性能退化"""
    if len(monitor.history) < window_size * 2:
        print("数据不足，无法检测性能退化")
        return False

    recent_rmse = [h['metrics'].get('rmse', float('inf'))
                  for h in monitor.history[-window_size:]]
    early_rmse = [h['metrics'].get('rmse', float('inf'))
                 for h in monitor.history[-window_size*2:-window_size]]

    recent_avg = np.mean(recent_rmse)
    early_avg = np.mean(early_rmse)
    degradation = (recent_avg - early_avg) / early_avg

    print(f"性能退化检测:")
    print(f"  早期平均RMSE: {early_avg:.4f}")
    print(f"  近期平均RMSE: {recent_avg:.4f}")
    print(f"  退化率: {degradation:.2%}")

    if degradation > threshold:
        print("⚠️  警告: 检测到性能退化")
        print("建议:")
        print("  1. 重新评估模型性能")
        print("  2. 更新权重配置")
        print("  3. 考虑引入新模型")
        return True
    else:
        print("✓  性能稳定")
        return False

# 使用示例
detect_performance_degradation(monitor)
```

import time

def profile_fusion(engine, models, config, true_values=None, n_runs=10):
    """性能分析"""
    times = []

    for i in range(n_runs):
        start_time = time.perf_counter()
        result = engine.fuse(models, config=config, true_values=true_values)
        end_time = time.perf_counter()
        times.append(end_time - start_time)

        if i == 0:
            # 第一次运行包含初始化，单独处理
            first_run_time = times[0]
            times = times[1:]

    avg_time = np.mean(times)
    std_time = np.std(times)
    min_time = np.min(times)
    max_time = np.max(times)

    print("性能分析结果:")
    print(f"  首次运行时间: {first_run_time:.4f}秒")
    print(f"  平均时间: {avg_time:.4f}秒")
    print(f"  标准差: {std_time:.4f}秒")
    print(f"  最小时间: {min_time:.4f}秒")
    print(f"  最大时间: {max_time:.4f}秒")
    print(f"  吞吐量: {1/avg_time:.2f} 次/秒")

    return {
        'first_run': first_run_time,
        'avg': avg_time,
        'std': std_time,
        'min': min_time,
        'max': max_time,
        'throughput': 1/avg_time
    }

# 使用示例
profile_fusion(engine, models, config, true_values)
